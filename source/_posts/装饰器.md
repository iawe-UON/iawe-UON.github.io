1、函数装饰器（不带参数）
```python
def my_decorator(func):
    def wrapper(*args, **kwargs):
        print("函数执行前的一些操作")
        result = func(*args, **kwargs)
        print("函数执行后的一些操作")
        return result
    return wrapper

@my_decorator
def say_hello(name):
    print(f"Hello, {name}!")
    return f"status:success"

# 使用装饰器
result = say_hello("Alice")
print(result)
```

2、函数装饰器（带参数）

```python
def repeats(times):
    def decorator(func):
        def wrapper(*args, **kwargs):
            for i in range(times):
                print(f"第{i+1}次输出：\n")
                result = func(*args, **kwargs)
            return result
        return wrapper
    return decorator

@repeats(times=3)
def say_hello():
    print("Hello World!")

say_hello()
```
